**要求：1、已知线性表中的元素以值递增有序的方式排列，所以单链表作为一个存储结构，
2、设计一个高效的算法，删除表中所有值大于mink且小于maxk的元素，
3、同时释放被删除的节点空间**

```C
#include <iostream>
#include <stdio.h>
#include <string>
typedef int elemtype;
typedef struct LNode {
	elemtype data;
	struct LNode* next;
}LNode, * Linklist;

typedef struct node
{
	elemtype data;
	struct node* next;//指针域
}LNode, * Linklist;

//分析：由表头开始从前往后遍历链表,没走过一个节点，看这个节点是否在mink和maxk之间，
//如果在此区间就删除节点。注意，在删除操作时绝对不能断链，其方法是用一个指针作为被删除节点的直接前驱。
bool DeleteMiddleElem(Linklist& L, int mink, int maxk)
{
	//删除单链表中大于mnk且小于maxk之间的数据元素
	Linklist p, q, prev = NULL;
	if (mink > maxk) return false;
	p = L;//L是头结点，p指向头结点
	prev = p;//prev也从头结点开始
	p = p->next;//让prev作为被删除节点的直接前驱
	while (p && p->data < maxk)
	{
		//先判断p->data的值是否也小于mink，如果小于mink的话就找到大于mink的结点
		if(p->data <= mink)
		{
			prev = p;
			p = p->next;
		}
		if (p->data > mink)
		{
			//删除p->data的结点
			prev->next = p->next;
			q = p;
			p = p->next;
			free(q);
		}
	}
	return 0;
}
```
